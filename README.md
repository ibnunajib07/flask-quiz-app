INSTALLATION
$ git clone https://github.com/ibnunajib07/flask-quiz-app.git

# Go to the directory
$ cd flask-quiz-app

# Use virtual env if you want
$ virtualenv ENV && source ENV/bin/activate

# Install requirement
$ pip install -r requirements.txt

# Export flask app and run
$ set FLASK_APP=main.py
$ flask run
